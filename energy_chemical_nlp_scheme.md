### 2.3 关键词提取

#### 2.3.1 多算法融合方案
```python
class KeywordExtractor:
    """多算法融合关键词提取"""
    
    def extract(self, text, top_k=10):
        # 1. TF-IDF (统计特征)
        tfidf_keywords = self.tfidf_extract(text, top_k)
        
        # 2. TextRank (图算法)
        textrank_keywords = self.textrank_extract(text, top_k)
        
        # 3. 领域词频 (先验知识)
        domain_keywords = self.domain_extract(text, top_k)
        
        # 4. 融合排序
        return self.fusion_rank(tfidf_keywords, 
                               textrank_keywords, 
                               domain_keywords)
```

#### 2.3.2 关键词权重计算
```
最终权重 = α·TF-IDF分数 + β·TextRank分数 + γ·领域重要性

其中:
    α = 0.3 (统计重要性)
    β = 0.3 (语义重要性)
    γ = 0.4 (领域先验)

领域重要性词典:
    减产协议: 1.0
    库存数据: 0.95
    OPEC会议: 0.95
    地缘冲突: 0.9
    需求增长: 0.85
    ...
```

```python
class SentimentClassifier(nn.Module):
    """基于BERT的情感分类器"""
    
    def __init__(self):
        self.bert = BertModel.from_pretrained("bert-base-chinese")
        self.dropout = nn.Dropout(0.1)
        self.classifier = nn.Linear(768, 3)  # 3分类
        
    def forward(self, input_ids, attention_mask):
        outputs = self.bert(input_ids, attention_mask)
        pooled = outputs.pooler_output
        pooled = self.dropout(pooled)
        logits = self.classifier(pooled)
        return logits
```

### 3.2 情感强度评分

#### 3.2.1 强度量化方案
```python
class SentimentIntensityScorer:
    """情感强度评分器"""
    
    def calculate_intensity(self, text, sentiment):
        """
        返回: 强度分数 (0-100)
        0-30: 弱
        30-60: 中等
        60-80: 强
        80-100: 极强
        """
        score = 50  # 基础分
        
        # 1. 程度词加成
        for word, weight in intensity_words.items():
            if word in text:
                score *= weight
        
        # 2. 情感词密度
        sentiment_density = self.count_sentiment_words(text) / len(text)
        score += sentiment_density * 20
        
        # 3. 标点符号加成
        if "！" in text or "!!" in text:
            score += 10
        
        # 4. 数据支撑加成
        if self.has_data_support(text):
            score += 15
        
        return min(100, max(0, score))
```

### 3.3 品种维度情感分析

#### 3.3.1 多品种情感提取
```python
class MultiProductSentimentAnalyzer:
    """多品种情感分析器"""
    
    def analyze(self, text):
        """
        分析文本中各品种的情感倾向
        返回: {品种: {"sentiment": 极性, "intensity": 强度, "confidence": 置信度}}
        """
        results = {}
        
        # 1. 识别文本中提到的品种
        products = self.extract_products(text)
        
        # 2. 对每个品种进行情感分析
        for product in products:
            # 提取与品种相关的句子
            relevant_sentences = self.extract_product_sentences(text, product)
            
            # 分析情感
            sentiment = self.classify_sentiment(relevant_sentences)
            intensity = self.calculate_intensity(relevant_sentences, sentiment)
            confidence = self.calculate_confidence(relevant_sentences)
            
            results[product] = {
                "sentiment": sentiment,
                "intensity": intensity,
                "confidence": confidence
            }
        
        return results
```

### 4.2 事件重要性评估

#### 4.2.1 重要性评分模型
```python
class EventImportanceEvaluator:
    """事件重要性评估器"""
    
    def evaluate(self, event):
        """
        综合评分 (0-100)
        """
        scores = {
            # 1. 事件类型权重
            "type_weight": self.get_type_weight(event.type),
            
            # 2. 影响范围
            "scope_score": self.evaluate_scope(event),
            
            # 3. 数据支撑
            "data_score": self.evaluate_data_support(event),
            
            # 4. 来源可信度
            "source_score": self.evaluate_source(event.source),
            
            # 5. 时效性
            "timeliness_score": self.evaluate_timeliness(event)
        }
        
        # 加权计算
        weights = {
            "type_weight": 0.25,
            "scope_score": 0.25,
            "data_score": 0.20,
            "source_score": 0.15,
            "timeliness_score": 0.15
        }
        
        final_score = sum(scores[k] * weights[k] for k in weights)
        return final_score
```

### 4.3 事件关联分析

#### 4.3.1 事件关联图谱
```python
class EventRelationAnalyzer:
    """事件关联分析器"""
    
    def build_event_graph(self, events):
        """
        构建事件关联图谱
        """
        graph = nx.DiGraph()
        
        for event in events:
            graph.add_node(event.id, **event.to_dict())
        
        # 建立关联
        for i, event1 in enumerate(events):
            for event2 in events[i+1:]:
                relation = self.detect_relation(event1, event2)
                if relation:
                    graph.add_edge(event1.id, event2.id, 
                                 relation_type=relation.type,
                                 strength=relation.strength)
        
        return graph
    
    def detect_relation(self, event1, event2):
        """
        检测两个事件之间的关系
        """
        # 1. 因果关系
        if self.is_causal(event1, event2):
            return Relation("CAUSAL", strength=0.9)
        
        # 2. 时间关系
        if self.is_temporal(event1, event2):
            return Relation("TEMPORAL", strength=0.7)
        
        # 3. 共现关系
        if self.is_cooccurring(event1, event2):
            return Relation("COOCCURRENCE", strength=0.5)
        
        return None
```

#### 5.2.2 维度映射算法
```python
class DimensionMapper:
    """多维度映射器"""
    
    def map_to_dimensions(self, text, entities, keywords, events):
        """
        将新闻映射到五个分析维度
        返回: {维度: {"score": 分数, "confidence": 置信度, "keywords": [关键词]}}
        """
        dimension_scores = {}
        
        for dimension in ["FUNDAMENTAL", "MACRO", "SENTIMENT", "TECHNICAL", "GEOPOLITICAL"]:
            # 1. 关键词匹配得分
            keyword_score = self.match_keywords(text, dimension_keywords[dimension])
            
            # 2. 实体类型匹配得分
            entity_score = self.match_entities(entities, dimension)
            
            # 3. 事件类型匹配得分
            event_score = self.match_events(events, dimension)
            
            # 4. 综合得分
            total_score = (keyword_score * 0.4 + 
                          entity_score * 0.3 + 
                          event_score * 0.3)
            
            # 5. 置信度计算
            confidence = self.calculate_confidence(keyword_score, entity_score, event_score)
            
            dimension_scores[dimension] = {
                "score": total_score,
                "confidence": confidence,
                "keywords": self.extract_dimension_keywords(text, dimension)
            }
        
        return dimension_scores
```

### 5.3 维度权重计算

#### 5.3.1 动态权重模型
```python
class DimensionWeightCalculator:
    """维度权重计算器"""
    
    def calculate_weights(self, news_item, market_context):
        """
        根据市场环境和新闻特征动态计算维度权重
        """
        # 基础权重
        base_weights = {
            "FUNDAMENTAL": 0.30,
            "MACRO": 0.20,
            "SENTIMENT": 0.15,
            "TECHNICAL": 0.15,
            "GEOPOLITICAL": 0.20
        }
        
        # 根据市场环境调整
        adjustments = self.get_market_adjustments(market_context)
        
        # 根据新闻特征调整
        news_adjustments = self.get_news_adjustments(news_item)
        
        # 合并调整
        final_weights = {}
        for dim in base_weights:
            final_weights[dim] = base_weights[dim] * (1 + adjustments.get(dim, 0)) * \
                                (1 + news_adjustments.get(dim, 0))
        
        # 归一化
        total = sum(final_weights.values())
        return {k: v/total for k, v in final_weights.items()}
    
    def get_market_adjustments(self, context):
        """根据市场环境获取调整系数"""
        adjustments = {}
        
        # 地缘紧张时期
        if context.geopolitical_tension > 0.7:
            adjustments["GEOPOLITICAL"] = 0.3
            adjustments["FUNDAMENTAL"] = -0.1
        
        # 宏观数据密集期
        if context.macro_data_density > 0.7:
            adjustments["MACRO"] = 0.2
        
        # 技术突破期
        if context.technical_breakout:
            adjustments["TECHNICAL"] = 0.2
        
        return adjustments
```
